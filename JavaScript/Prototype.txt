// 🔎 What "Prototype" Means
// A prototype is an object that other objects can inherit properties/methods from.
// Think of it as a shared blueprint or fallback storage.

// ⚙️ What It Does
// - Every object has an internal [[Prototype]] link (accessible via __proto__).
// - Property lookup goes through the prototype chain until found or returns undefined.

// 📌 Why We Need It
// - Memory efficiency: methods shared instead of duplicated.
// - Inheritance: reuse behavior without rewriting.
// - Extensibility: add new methods to prototypes (even built-ins).

// 🧩 Example
function Car(brand) {
  this.brand = brand;
}

// Add a method to the prototype
Car.prototype.drive = function() {
  console.log(`${this.brand} is driving...`);
};

const tesla = new Car("Tesla");
const toyota = new Car("Toyota");

tesla.drive();   // Tesla is driving...
toyota.drive();  // Toyota is driving...

// 🚀 Prototype Chain Visualization
// tesla.__proto__ === Car.prototype
// Car.prototype.__proto__ === Object.prototype
// Object.prototype.__proto__ === null
