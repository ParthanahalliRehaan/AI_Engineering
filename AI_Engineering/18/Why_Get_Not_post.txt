// Why you see GET requests in Worker logs
// ---------------------------------------
// - Browsers auto-fetch /favicon.ico with GET, even if your main request was POST when you make a request directly at '/' ie root even if you make a POST req at '/' its considered as GET.
// - Some clients/monitors probe the root `/` with GET to check connectivity.
// - CORS preflight (OPTIONS) and incidental GETs appear alongside your POST.
// - Cloudflare Worker observability logs *all* requests hitting the Worker, not just POSTs.

// Worker logs vs. AI Gateway logs
// -------------------------------
// - Worker logs: raw HTTP traffic (GET, POST, OPTIONS, etc.) hitting your Worker.
// - AI Gateway logs: structured analytics about AI API usage (prompt, response size, latency).
// - They are NOT the same. Worker logs = traffic records; Gateway logs = usage metrics.

// Confirming POSTs at root `/`
// ----------------------------
// If you POST to `/`, Cloudflare logs it as POST. The method is preserved.
// Example entry: Method: POST, URL: https://your-worker.example.com/

// Example Worker code to log only POST requests
export default {
  async fetch(request) {
    if (request.method === "POST") {
      console.log("POST received at:", request.url);
    } else {
      console.log("Ignoring non-POST:", request.method, request.url);
    }
    return new Response("ok");
  }
};

/**
 * ============================================================================
 * HOW WE SOLVED THE CLOUDFLARE WORKER "GET REQUEST" ISSUE
 * ============================================================================
 * * 1. Diagnosed the "Open Door" Problem: 
 * The Worker was previously accepting all traffic at the root URL (/). 
 * Automated bots and browser `/favicon.ico` checks were hitting the server 
 * with GET requests, crashing the script when it tried to parse JSON.
 * * 2. Created a Dedicated API Route: 
 * We moved the endpoint from the root `/` to a specific path: `/api/gift`. 
 * This physically separates web app traffic from random internet noise.
 * * 3. Implemented Method Filtering (The Bouncer): 
 * We added `if (request.method === 'POST')`. This instructs the server to 
 * ignore standard GET requests and only execute the OpenAI code when data 
 * is explicitly submitted.
 * * 4. Updated the Frontend (script.js): 
 * We changed the fetch() URL to point directly to the new route:
 * `fetch("https://openai-api-worker.parthanahalli-rehaan.workers.dev/api/gift", { method: "POST" ... })`
 * * 5. Leveraged Real-Time Logs: 
 * We added `console.log()` statements to track incoming methods and URLs 
 * so they show up clearly in Cloudflare's "Live" logs.
 * ============================================================================
 */

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Methods': 'POST, OPTIONS',
  'Access-Control-Allow-Headers': 'Content-Type'
};

import OpenAI from "openai";

export default {
  async fetch(request, env, ctx) {
    // 1. Handle CORS Preflight requests
    if (request.method === 'OPTIONS') {
      return new Response(null, { headers: corsHeaders });
    }

    const url = new URL(request.url);

    // 2. Route the request to our specific endpoint (Checking Path AND Method)
    if (url.pathname === '/api/gift' && request.method === 'POST') {
      try {
        const { prompt } = await request.json();
        
        // This log will show up in Cloudflare's "Live" logs!
        console.log(`[POST /api/gift] Received prompt: "${prompt}"`);

        const openai = new OpenAI({ 
          apiKey: env.OPENAI_API_KEY, 
          baseURL:"https://gateway.ai.cloudflare.com/v1/d7297e3e5152a44ac5c13eaea4026f3d/openai-api/compat/chat/completions" 
        });

        const messages = [
          {
            role: "system",
            content: `You are the Gift Genie!
            Make your gift suggestions thoughtful and practical.
            Your response must be under 100 words. 
            Skip intros and conclusions. 
            Only output gift suggestions.`,
          },
          { role: "user", content: prompt },
        ];

        const response = await openai.chat.completions.create({
          model: env.OPENAI_AI_MODEL,
          messages,
        });

        console.log("[POST /api/gift] Successfully generated response from OpenAI");

        return new Response(
          JSON.stringify({ reply: response.choices[0].message.content }),
          { 
            status: 200,
            headers: { ...corsHeaders, 'Content-Type': 'application/json' } 
          }
        );
      } catch (e) {
        console.error("[POST /api/gift] Error:", e.message);
        return new Response(
          JSON.stringify({ error: e.message }),
          { 
            status: 500,
            headers: { ...corsHeaders, 'Content-Type': 'application/json' } 
          }
        );
      }
    }

    // 3. Catch-all for any other requests (like random GETs or wrong paths)
    return new Response(
      JSON.stringify({ error: "Not Found. Please POST to /api/gift" }), 
      { 
        status: 404, 
        headers: { ...corsHeaders, 'Content-Type': 'application/json' } 
      }
    );
  },
};
