## Step 3 — Update `config.mjs`

```javascript
import openai from "openai";
import { HfInference } from "@huggingface/inference";
import path from "path";
import { fileURLToPath } from "url";
import dotenv from "dotenv";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
dotenv.config({ path: path.resolve(__dirname, "../.env") });

if (!process.env.AI_KEY) {
    throw new Error("OpenAI API key is missing or invalid.");
}
if (!process.env.HF_API_KEY) {
    throw new Error("HuggingFace API key is missing or invalid.");
}

export const openaiClient = new openai({
    apiKey: process.env.AI_KEY,
    baseURL: process.env.AI_URL
});

export const hf = new HfInference(process.env.HF_API_KEY);
```

> ⚠️ Notice — `export default` changed to **named exports** (`export const`) because now you're exporting two clients.

## Step 4 — `script.js`

```javascript
import { openaiClient, hf } from "./config.mjs";
import path from "path";
import { fileURLToPath } from "url";
import dotenv from "dotenv";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
dotenv.config({ path: path.resolve(__dirname, "../.env") });

const embedIt = "Cat";

/*
Can be Array's

const embedIt = [
  "Beyond Mars: speculating life on distant planets.",
  "Jazz under stars: a night in New Orleans' music scene.",
  "Mysteries of the deep: exploring uncharted ocean caves.",
  "Rediscovering lost melodies: the rebirth of vinyl culture.",
  "Tales from the tech frontier: decoding AI ethics.",
];
*/

async function main() {
    const response = await hf.featureExtraction({
        model: process.env.EMBED_MODEL,
        inputs: embedIt,
    });

    // Normalize — HF returns flat array for single string, nested for batch
    const embeddings = typeof embedIt === "string" ? [response] : response;

    console.log(response);                  // raw output
    console.log(embeddings[0]);             // first vector  (like embedding.data[0].embedding)
    console.log(embeddings[0].length);      // dimensions    (like embedding.data[0].embedding.length)
}

main();
```

## How imports changed — important to note

**Before** (single default export):
```javascript
// config.mjs
export default new openai({...});

// script.js
import openai from "./config.mjs";
```

**After** (named exports — two clients):
```javascript
// config.mjs
export const openaiClient = new openai({...});
export const hf = new HfInference(...);

// script.js
import { openaiClient, hf } from "./config.mjs";
```

This way your OpenAI client is still there untouched for chat completions — only embeddings now go through HuggingFace.
