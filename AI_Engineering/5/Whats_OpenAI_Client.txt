// Import the OpenAI SDK class
// "OpenAI" = the class (blueprint)
const openAI = new OpenAI({
    apiKey: process.env.AI_KEY,   // Authentication token
    baseURL: process.env.AI_URL   // API endpoint (default or custom)
});

// "openAI" = the object (instance) created from the class.
// This object is your configured client for sending requests to models.

// Example: using the client to call a model
const response = await openAI.chat.completions.create({
    model: "gpt-4.1",  // The actual model you want to use
    messages: [
        { role: "system", content: "You are a helpful assistant." },
        { role: "user", content: "Explain the difference between a class and an object." }
    ]
});

console.log(response.choices[0].message.content);

// Summary:
// - OpenAI → class provided by the SDK.
// - openAI → object (instance) created from that class.
// - Models (like GPT-4.1) → what you actually query through the client.
