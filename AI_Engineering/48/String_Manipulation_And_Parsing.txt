## 🔤 Basic String Operations in JavaScript

### 1. **Creating Strings**
```js
let str1 = "Hello World";
let str2 = 'Single quotes work too';
let str3 = `Template literals allow ${str1}`;
```

---

### 2. **Accessing Characters**
```js
let text = "Action: get_weather";
console.log(text[0]);      // "A"
console.log(text.charAt(7)); // ":"
```

---

### 3. **Length of a String**
```js
console.log(text.length); // 20
```

---

### 4. **Searching in Strings**
```js
console.log(text.includes("Action")); // true
console.log(text.indexOf("get"));     // 8
console.log(text.startsWith("Action")); // true
console.log(text.endsWith("weather"));  // true
```

---

### 5. **Extracting Substrings**
```js
let actionLine = "Action: get_weather(location=Paris)";
console.log(actionLine.slice(8)); // "get_weather(location=Paris)"
console.log(actionLine.substring(8, 18)); // "get_weather"
```

---

### 6. **Splitting Strings**
This is crucial for parsing GPT responses separated by `\n`.

```js
let response = "Thought: I should look up the weather\nAction: get_weather(location=Paris)\nPause: waiting";
let lines = response.split("\n");
console.log(lines);
// ["Thought: I should look up the weather", "Action: get_weather(location=Paris)", "Pause: waiting"]
```

---

### 7. **Trimming Whitespace**
```js
let messy = "   Action: get_weather   ";
console.log(messy.trim()); // "Action: get_weather"
```

---

### 8. **Replacing Text**
```js
let updated = actionLine.replace("Paris", "Tokyo");
console.log(updated); // "Action: get_weather(location=Tokyo)"
```

---

### 9. **Regex Basics (for Parsing)**
Regex helps extract function names and parameters.

```js
let actionLine = "Action: get_weather(location=Paris)";
let regex = /Action:\s*(\w+)\((.*)\)/;
let match = actionLine.match(regex);

console.log(match[1]); // "get_weather" (function name)
console.log(match[2]); // "location=Paris" (parameters)
```

## 🔄 Putting It Together: Parsing GPT Output

Here’s a **minimal working example** of the loop you were learning about:

```js
function processResponse(response) {
  // Split response into lines
  let lines = response.split("\n");

  // Look for "Action:" line
  let actionLine = lines.find(line => line.startsWith("Action:"));

  if (actionLine) {
    // Extract function + parameters using regex
    let regex = /Action:\s*(\w+)\((.*)\)/;
    let match = actionLine.match(regex);

    if (match) {
      let funcName = match[1];
      let params = match[2];
      console.log("Function:", funcName);
      console.log("Parameters:", params);

      // Simulate function call
      let observation = `Observation: ${funcName} returned result for ${params}`;
      console.log(observation);

      // Loop again with observation appended
      return processResponse("Thought: Done\n" + observation);
    }
  } else {
    // No action → final answer
    console.log("Final Answer:", lines[0]);
  }
}

// Example run
let gptResponse = "Thought: I should look up the weather\nAction: get_weather(location=Paris)\nPause: waiting";
processResponse(gptResponse);
```

---

## 🧾 Key Takeaways
- Use `split("\n")` to break GPT’s response into lines.
- Use `find()` or `filter()` to locate the `"Action:"` line.
- Use `regex` to extract function names and parameters.
- Build a loop: if `"Action:"` exists → call function → add observation → loop again. If not → stop.
