# Step 1: Stop tracking the .env file (but keep it locally)
git rm --cached AI_Engineering/.env

# Step 2: Upgrade pip (recommended before installing new Python packages)
python -m pip install --upgrade pip

# Step 3: Install git-filter-repo (modern tool for rewriting Git history)
python -m pip install git-filter-repo

# Step 4: Check available options for git-filter-repo
git filter-repo --help

# Step 5: Rewrite history to remove AI_Engineering/.env from all commits
# --path specifies the file, --invert-paths means "remove this file from history"
git filter-repo --path AI_Engineering/.env --invert-paths

# --- Alternative method (older, less recommended) ---
# If git-filter-repo is not available, you can use git filter-branch:
git filter-branch --force --index-filter \
  "git rm --cached --ignore-unmatch AI_Engineering/.env" \
  --prune-empty --tag-name-filter cat -- --all
