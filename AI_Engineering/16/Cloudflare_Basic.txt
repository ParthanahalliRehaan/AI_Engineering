Theory
  // Cloudflare Workers are lightweight serverless functions
  // that run on Cloudflare's edge network. They let you deploy
  // JavaScript (or TypeScript) without managing servers.

  // Wrangler is Cloudflare's CLI tool used to:
  // - Generate new Worker projects
  // - Configure deployment settings
  // - Publish Workers to Cloudflare's global network

  // Example Worker (hello world):
  export default {
    async fetch(request) {
      return new Response("Hello World", { status: 200 });
    }
  }

                # Use npm to scaffold a new Cloudflare Worker project
                npm create cloudflare@latest my-worker

                # During setup, choose:
                # - "Worker only" template (simplest)
                # - JavaScript (or TypeScript if you prefer)
                # - Name: my-worker (or any name you like)

                # Move into the project directory
                cd my-worker
CLIs
  # Deploys
  npx wrangler deploy

  # Authenticate Wrangler with your Cloudflare account
  wrangler login

  # Run the Worker locally for testing
  wrangler dev

  # Publish the Worker to Cloudflare's edge network
  wrangler publish

  # Check account info
  wrangler whoami

  # Update Wrangler configuration
  wrangler config
